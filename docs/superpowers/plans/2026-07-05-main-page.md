# AI E-Book Generator — Main App Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the full-stack main library page — FastAPI backend with SQLite storing books and credits, Next.js 15 frontend with the sidebar shell, book grid, book cards, and new-book drawer — all wired together and working in the browser.

**Architecture:** FastAPI serves a REST API (`/api/books`, `/api/credits`) backed by SQLite via SQLModel. Next.js 15 App Router renders the library as a server component that fetches initial data; interactive pieces (drawer, card actions, filters) are client components. Design tokens live in global CSS; each component owns its scoped styles via CSS Modules.

**Tech Stack:** Python 3.11+, FastAPI 0.115+, SQLModel 0.0.21+, uvicorn · Next.js 15, React 19, TypeScript 5, CSS Modules

```python
from typing import List
from fastapi import APIRouter, Depends, HTTPException
from sqlmodel import Session, select
from ..database import get_session
from ..models.book import Book
from ..schemas.book import BookCreate, BookUpdate

router = APIRouter()


@router.get("/", response_model=List[Book])
def list_books(session: Session = Depends(get_session)):
    return session.exec(select(Book).order_by(Book.created_at.desc())).all()


@router.post("/", response_model=Book, status_code=201)
def create_book(data: BookCreate, session: Session = Depends(get_session)):
    book = Book(**data.model_dump())
    session.add(book)
    session.commit()
    session.refresh(book)
    return book


@router.get("/{book_id}", response_model=Book)
def get_book(book_id: int, session: Session = Depends(get_session)):
    book = session.get(Book, book_id)
    if not book:
        raise HTTPException(404, "Book not found")
    return book


@router.patch("/{book_id}", response_model=Book)
def update_book(book_id: int, data: BookUpdate, session: Session = Depends(get_session)):
    book = session.get(Book, book_id)
    if not book:
        raise HTTPException(404, "Book not found")
    for k, v in data.model_dump(exclude_unset=True).items():
        setattr(book, k, v)
    session.commit()
    session.refresh(book)
    return book


@router.delete("/{book_id}", status_code=204)
def delete_book(book_id: int, session: Session = Depends(get_session)):
    book = session.get(Book, book_id)
    if not book:
        raise HTTPException(404, "Book not found")
    session.delete(book)
    session.commit()
```

from contextlib import asynccontextmanager
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from .database import create_db

@asynccontextmanager
async def lifespan(app: FastAPI):

import type { Book, BookCreate, Credits } from './types'

const BASE = 'http://localhost:8000/api'

async function req<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${BASE}${path}`, {
    headers: { 'Content-Type': 'application/json' },
    ...init,
  })
  if (!res.ok) throw new Error(`API ${res.status}: ${path}`)
  if (res.status === 204) return undefined as T
  return res.json()
}

export const api = {
  books: {
    list: ()                          => req<Book[]>('/books/'),
    create: (data: BookCreate)        => req<Book>('/books/', { method: 'POST', body: JSON.stringify(data) }),
    get: (id: number)                 => req<Book>(`/books/${id}`),
    update: (id: number, patch: Partial<Book>) =>
      req<Book>(`/books/${id}`, { method: 'PATCH', body: JSON.stringify(patch) }),
    delete: (id: number)              => req<void>(`/books/${id}`, { method: 'DELETE' }),
  },
  credits: {
    get: ()      => req<Credits>('/credits/'),
    consume: ()  => req<Credits>('/credits/consume', { method: 'POST' }),
  },
}

const COVER_GRADIENTS: Record<number, string> = {
  264: 'linear-gradient(160deg, #1a0a38 0%, #5b21b6 60%, #7c3aed 100%)',
  155: 'linear-gradient(160deg, #0a1f12 0%, #065f46 60%, #10b981 100%)',

  175: 'linear-gradient(160deg, #001010 0%, #115e59 60%, #14b8a6 100%)',
}

function getCoverGradient(hue: number): string {
  const key = Object.keys(COVER_GRADIENTS)
    .map(Number)
    .reduce((a, b) => (Math.abs(b - hue) < Math.abs(a - hue) ? b : a))
  return COVER_GRADIENTS[key]
}

const TYPE_LABELS: Record<string, string> = {
  childrens: "Children's",
  tutorial: 'Tutorial',
  guide: 'Guide',
  educational: 'Educational',

}

  book: Book
  onFavoriteToggle: (id: number, val: boolean) => void
  onRead: (id: number) => void
  onDelete: (id: number) => void
}

export function BookCard({ book, onFavoriteToggle, onRead, onDelete }: Props) {
  const gradient = getCoverGradient(book.cover_hue)
  const isGenerating = book.status === 'generating'
  const isError = book.status === 'error'

  async function toggleFav(e: React.MouseEvent) {
    e.stopPropagation()
    await api.books.update(book.id, { favorite: !book.favorite })
    onFavoriteToggle(book.id, !book.favorite)
  }

    <div className={s.card}>
      {/* Cover */}

```tsx
'use client'

import { useState, useMemo } from 'react'
import { useRouter } from 'next/navigation'
import { Book, BookCreate, Credits } from '@/lib/types'
import { api } from '@/lib/api'
import { Sidebar } from '@/components/sidebar/sidebar'
import { Topbar } from '@/components/topbar/topbar'
import { BookGrid } from '@/components/book-grid/book-grid'
import { NewBookDrawer } from '@/components/new-book-drawer/new-book-drawer'
import s from './library.module.css'

interface Props {
  initialBooks: Book[]
  initialCredits: Credits
}

export function LibraryClient({ initialBooks, initialCredits }: Props) {
  const router = useRouter()
  const [books, setBooks] = useState<Book[]>(initialBooks)
  const [credits, setCredits] = useState<Credits>(initialCredits)
  const [drawerOpen, setDrawerOpen] = useState(false)
  const [search, setSearch] = useState('')
  const [activeNav, setActiveNav] = useState('all')

  const counts = useMemo(() => ({
    all:        books.length,
    favorites:  books.filter(b => b.favorite).length,
    generating: books.filter(b => b.status === 'generating').length,
  }), [books])

  const visible = useMemo(() => {
    let list = books
    if (search) {
      const q = search.toLowerCase()
      list = list.filter(b => b.title.toLowerCase().includes(q) || b.prompt.toLowerCase().includes(q))
    }
    if (activeNav === 'favorites')  list = list.filter(b => b.favorite)
    if (activeNav === 'generating') list = list.filter(b => b.status === 'generating')
    if (['childrens','tutorial','guide','educational','story','custom'].includes(activeNav)) {
      list = list.filter(b => b.book_type === activeNav)
    }
    return list
  }, [books, search, activeNav])

  async function handleCreate(data: BookCreate) {
    const consumed = await api.credits.consume()
    setCredits(consumed)
    const book = await api.books.create(data)
    setBooks(prev => [book, ...prev])
    // In a real app, kick off generation here via WebSocket or polling
  }

  function handleFavoriteToggle(id: number, val: boolean) {
    setBooks(prev => prev.map(b => b.id === id ? { ...b, favorite: val } : b))
  }

  function handleRead(id: number) {
    router.push(`/book/${id}`)
  }

  async function handleDelete(id: number) {
    await api.books.delete(id)
    setBooks(prev => prev.filter(b => b.id !== id))
  }

  return (
    <>
      <Sidebar active={activeNav} bookCounts={counts} credits={credits} />

      <div className={s.main}>
        <Topbar
          onSearch={setSearch}
          onFormatChange={() => {}}
          onViewChange={() => {}}
          onNewBook={() => setDrawerOpen(true)}
        />

        <div className={s.content}>
          <div className={s.pageHeader}>
            <span className={s.pageTitle}>My Library</span>
            <span className={s.pageCount}>{visible.length} books</span>
          </div>

          <BookGrid
            books={visible}
            onNewBook={() => setDrawerOpen(true)}
            onFavoriteToggle={handleFavoriteToggle}
            onRead={handleRead}
            onDelete={handleDelete}
          />
        </div>
      </div>

      {drawerOpen && (
        <NewBookDrawer
          onClose={() => setDrawerOpen(false)}
          onSubmit={handleCreate}
        />
      )}
    </>
  )
}
```
